   public static int getSum(int b, int n1, int n2){
       // write ur code here
       
       int decn1=basetodec(b,n1);
       int decn2=basetodec(b,n2);
       int sum =decn1+decn2;         // sum also works fine

   public static int basetodec(int b,int n)  //this function works fine
   {
       int rem=0;
       int multi=1;
       int ans=0;
       while(n!=0)
       { 
           rem=n%10;
           ans=ans+rem*multi;
           multi=multi*b;
           n=n/10;
         
       } 
       //System.out.println(ans);
       return (ans);
   }
   
   public static int dectobase(int b,int n)
   {
      int ans=0;
      int pow=1;
      int rem=0;
      while(n!=0)
      {
          rem=n%b;
          ans=ans+pow*rem;
          pow=pow*10;
          n=n/b;
      }
      return(ans);
   }
